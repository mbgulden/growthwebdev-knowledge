Three.js Shader Logic (pve6):

// pve6: Fog of War Fragment Shader Fragment

const fogOfWarShader = {

  uniforms: {

    tMask: { value: maskCanvasTexture }, // The white/black canvas

    tDiffuse: { value: terrainTexture },   // The high-res Nanobanana texture

  },

  fragmentShader: `

    varying vec2 vUv;

    uniform sampler2D tMask;

    uniform sampler2D tDiffuse;


    void main() {

      float mask = texture2D(tMask, vUv).r;

      vec4 color = texture2D(tDiffuse, vUv);

      

      // If mask is 0 (black), set opacity to 0 or dim it

      gl_FragColor = vec4(color.rgb, mask); 

      

      // Optional: Give 'visited' areas a grey tint

      if(mask < 0.1) gl_FragColor = vec4(color.rgb * 0.2, 1.0);

    }

  `

};
